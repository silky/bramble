    type MIME           = String
    type PathSegment    = String
    type TraitName      = String
    type ParamName      = String
    type ResponseCode   = Int

x   type RouteLookup    = Lookup PathSegment  RouteInfo
    type MimeLookup     = Lookup MIME         ResponseInfo
    type ResponseLookup = Lookup ResponseCode ResponseInfo
x   type MethodLookup   = Lookup Method       Info
x   type TraitLookup    = Lookup TraitName    Info
x   type QueryLookup    = Lookup ParamName    ParamSpec

x   data RamlFile = RamlFile
x   data RamlMetadata = RamlMetadata
x   data Method = Get | Put | Post | Delete | Custom String deriving (Eq,Ord,Show)
    data ParamType = StringParam | NumberParam  | IntegerParam
    data RouteInfo = RouteInfo
    data Info = Info
x   data ParamSpec = ParamSpec
    data ResponseInfo = ResponseInfo
    data ResponseInfo = ResponseInfo
x   data Schema = Schema String deriving (Eq,Ord,Show)
